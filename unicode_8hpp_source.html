170[[nodiscard]] inline auto operator==(const unicode_string& a, const unicode_string& b) -> bool
171{
172 if (a.size() != b.size()) {
173 return false;
174 }
175
176 for (unicode_string::size_type index = 0; index < a.size(); ++index) {
177 const auto aChar = a.at(index);
178 const auto bChar = b.at(index);
179 if (aChar != bChar) {
180 return false;
181 }
182 }
183
184 return true;
185}

188{
189 return !(a == b);
