350  typename point_traits<T>::value_type
351 {
352  if constexpr (basic_point<T>::isIntegral) {
353  const auto xDiff = std::pow(from.x() - to.x(), T{2});
354  const auto yDiff = std::pow(from.y() - to.y(), T{2});
355  const auto dist = std::sqrt(xDiff + yDiff);
356  return static_cast<int>(std::round(dist));
357  }
358  else {
359  return std::sqrt(std::pow(from.x() - to.x(), T{2}) + std::pow(from.y() - to.y(), T{2}));
360  }
