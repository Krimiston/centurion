131[[nodiscard]] constexpr auto is_user_event(const event_type type) noexcept -> bool
132{
133 const auto raw = to_underlying(type);
134 return raw >= SDL_USEREVENT && raw < SDL_LASTEVENT;
135}

352template <typename T>
353class event_base
354{
355 public:
356 explicit event_base(const event_type type)
357 {
358 set_timestamp(u32ms{SDL_GetTicks()});
359 set_type(type);
360 }
361
362 explicit event_base(const T& event) noexcept : mEvent{event} {}
363
364 explicit event_base(T&& event) noexcept : mEvent{std::move(event)} {}
365
366 void set_timestamp(const u32ms timestamp) noexcept(on_msvc)
367 {
368 mEvent.timestamp = timestamp.count();
369 }
370
371 void set_type(const event_type type) noexcept { mEvent.type = to_underlying(type); }
372
373 [[nodiscard]] auto timestamp() const -> u32ms { return u32ms{mEvent.timestamp}; }
374
375 [[nodiscard]] auto type() const noexcept -> event_type { return event_type{mEvent.type}; }
376
377 [[nodiscard]] auto get() const noexcept -> const T& { return mEvent; }
378
379 protected:
380 T mEvent{};
381};
