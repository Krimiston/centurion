172 {
173 const auto channel = Mix_PlayMusic(mMusic.get(), detail::max(iterations, forever));
174 if (channel != -1) {

197 {
198 assert(duration.count() > 0);
199 return Mix_FadeInMusic(mMusic.get(), detail::max(iterations, forever), duration.count()) ==
200 0;
201 }
202
203 static auto fade_out(const ms_type duration) noexcept(noexcept(duration.count())) -> result
204 {
205 assert(duration.count() > 0);

209 else {
210 return failure;
211 }
212 }

424template <typename T>
425class basic_sound_effect final
426{
427 public:
428 using channel_index = int;
429 using ms_type = millis<int>;
430
431 inline constexpr static channel_index undefined_channel = -1;
432 inline constexpr static int forever = -1;
433
434 explicit basic_sound_effect(maybe_owner<Mix_Chunk*> sound) noexcept(detail::is_handle<T>)
435 : mChunk{sound}
436 {
437 if constexpr (detail::is_owner<T>) {
438 if (!mChunk) {
439 throw mix_error{};
440 }
441 }
442 }
443
444 template <typename TT = T, detail::enable_for_owner<TT> = 0>
445 explicit basic_sound_effect(const char* file) : mChunk{Mix_LoadWAV(file)}
446 {
447 if (!mChunk) {
448 throw mix_error{};
449 }
450 }
451
452 template <typename TT = T, detail::enable_for_owner<TT> = 0>
453 explicit basic_sound_effect(const std::string& file) : basic_sound_effect{file.c_str()}
454 {}
455
456 template <typename TT = T, detail::enable_for_handle<TT> = 0>
457 explicit basic_sound_effect(const sound_effect& owner) noexcept : mChunk{owner.get()}
458 {}
459
460 auto play(const int iterations = 0) noexcept -> result
461 {
462 mChannel = Mix_PlayChannel(mChannel, mChunk.get(), detail::max(iterations, forever));
463 return mChannel != -1;
464 }
465
466 void stop() noexcept
467 {
468 if (is_playing()) {
469 Mix_Pause(mChannel);
470 mChannel = undefined_channel;
471 }
472 }
473
474 [[nodiscard]] auto is_playing() const noexcept -> bool
475 {
476 return (mChannel != undefined_channel) && Mix_Playing(mChannel);
477 }
478
479 template <typename TT = T, detail::enable_for_owner<TT> = 0>
480 [[nodiscard]] static auto is_any_playing() noexcept -> bool
481 {
482 return Mix_Playing(undefined_channel);
483 }
484
485 void fade_in(const ms_type duration) noexcept(noexcept(duration.count()))
486 {
487 assert(duration.count() > 0);
488 if (!is_playing()) {
489 mChannel = Mix_FadeInChannel(mChannel, get(), 0, duration.count());
490 }
491 }
492
493 void fade_out(const ms_type duration) noexcept(noexcept(duration.count())) // NOLINT
494 {
495 assert(duration.count() > 0);
496 if (is_playing()) {
497 Mix_FadeOutChannel(mChannel, duration.count());
498 }
499 }
500
501 [[nodiscard]] auto is_fading() const noexcept -> bool
502 {
503 return is_playing() && Mix_FadingChannel(mChannel);
504 }
505
506 [[nodiscard]] auto channel() const noexcept -> maybe<channel_index>
507 {
508 if (mChannel != undefined_channel) {
509 return mChannel;
510 }
511 else {
512 return nothing;
513 }
514 }
515
516 void set_volume(const int volume) noexcept
517 {
518 Mix_VolumeChunk(mChunk.get(), detail::clamp(volume, 0, max_volume()));
519 }
520
521 [[nodiscard]] auto volume() const noexcept -> int { return mChunk->volume; }
522
523 [[nodiscard]] constexpr static auto max_volume() noexcept -> int { return MIX_MAX_VOLUME; }
524
525 template <typename TT = T, detail::enable_for_owner<TT> = 0>
526 [[nodiscard]] static auto has_decoder(const char* name) noexcept -> bool
527 {
528 return Mix_HasChunkDecoder(name) == SDL_TRUE;
529 }
530
531 template <typename TT = T, detail::enable_for_owner<TT> = 0>
532 [[nodiscard]] static auto get_decoder(const int index) noexcept -> const char*
533 {
534 return Mix_GetChunkDecoder(index);
535 }
536
537 template <typename TT = T, detail::enable_for_owner<TT> = 0>
538 [[nodiscard]] static auto decoder_count() noexcept -> int
539 {
540 return Mix_GetNumChunkDecoders();
541 }
542
543 [[nodiscard]] auto get() const noexcept -> Mix_Chunk* { return mChunk.get(); }
544
545#if SDL_MIXER_VERSION_ATLEAST(2, 6, 0)
546
547 static void set_master_volume(const int volume) noexcept { Mix_MasterVolume(volume); }
548
549 [[nodiscard]] static auto master_volume() noexcept -> int { return Mix_MasterVolume(-1); }
550
551#endif // SDL_MIXER_VERSION_ATLEAST(2, 6, 0)
552
553 private:
554 detail::pointer<T, Mix_Chunk> mChunk;
555 channel_index mChannel{undefined_channel};
556
557#ifdef CENTURION_MOCK_FRIENDLY_MODE
558
559 public:
560 void set_channel(const channel_index channel) noexcept { mChannel = channel; }
561
562#endif // CENTURION_MOCK_FRIENDLY_MODE
563};

575#endif // CENTURION_HAS_FEATURE_FORMAT
576}
