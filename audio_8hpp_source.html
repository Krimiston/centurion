153class music final
154{
155 public:
156 using channel_index = int;
157 using ms_type = millis<int>;
158 using music_hook_callback = void (*)(void*, uint8*, int);
159
160 inline constexpr static int forever = -1;
161
162 explicit music(const char* file) : mMusic{Mix_LoadMUS(file)}
163 {
164 if (!mMusic) {
165 throw mix_error{};
166 }
167 }
168
169 explicit music(const std::string& file) : music{file.c_str()} {}
170
171 auto play(const int iterations = 0) noexcept -> std::optional<channel_index>
172 {
173 const auto channel = Mix_PlayMusic(mMusic.get(), detail::max(iterations, forever));
174 if (channel != -1) {
175 return channel;
176 }
177 else {
178 return std::nullopt;
179 }
180 }
181
182 static void resume() noexcept { Mix_ResumeMusic(); }
183
184 static void pause() noexcept { Mix_PauseMusic(); }
185
186 static void halt() noexcept
187 {
188 Mix_HaltMusic(); /* This appears to always return 0, so we ignore it */
189 }
190
191 [[nodiscard]] static auto is_playing() noexcept -> bool { return Mix_PlayingMusic(); }
192
193 [[nodiscard]] static auto is_paused() noexcept -> bool { return Mix_PausedMusic(); }
194
195 auto fade_in(const ms_type duration,
196 const int iterations = 0) noexcept(noexcept(duration.count())) -> result
197 {
198 assert(duration.count() > 0);
199 return Mix_FadeInMusic(mMusic.get(), detail::max(iterations, forever), duration.count()) ==
200 0;
201 }
202
203 static auto fade_out(const ms_type duration) noexcept(noexcept(duration.count())) -> result
204 {
205 assert(duration.count() > 0);
206 if (!is_fading()) {
207 return Mix_FadeOutMusic(duration.count()) != 0;
208 }
209 else {
210 return failure;
211 }
212 }
213
214 [[nodiscard]] static auto get_fade_status() noexcept -> fade_status
215 {
216 return static_cast<fade_status>(Mix_FadingMusic());
217 }
218
219 [[nodiscard]] static auto is_fading() noexcept -> bool
220 {
221 const auto status = get_fade_status();
222 return status == fade_status::in || status == fade_status::out;
223 }
224
225 [[nodiscard]] static auto is_fading_in() noexcept -> bool
226 {
227 return get_fade_status() == fade_status::in;
228 }
229
230 [[nodiscard]] static auto is_fading_out() noexcept -> bool
231 {
232 return get_fade_status() == fade_status::out;
233 }
234
235 static void rewind() noexcept { Mix_RewindMusic(); }
236
237 static auto set_position(const double position) noexcept -> result
238 {
239 return Mix_SetMusicPosition(position) == 0;
240 }
241
242 static void set_volume(const int volume) noexcept

244 Mix_VolumeMusic(detail::clamp(volume, 0, MIX_MAX_VOLUME));
245 }

365 {}

368 explicit basic_sound_effect(const sound_effect& owner) noexcept : mChunk{owner.get()}
369 {}
370
371 auto play(const int iterations = 0) noexcept -> result
372 {
373 mChannel = Mix_PlayChannel(mChannel, mChunk.get(), detail::max(iterations, forever));
374 return mChannel != -1;

386 {
387 return (mChannel != undefined_channel) && Mix_Playing(mChannel);
388 }

390 template <typename TT = T, detail::enable_for_owner<TT> = 0>

395
396 void fade_in(const ms_type duration) noexcept(noexcept(duration.count()))
397 {
398 assert(duration.count() > 0);
399 if (!is_playing()) {
400 mChannel = Mix_FadeInChannel(mChannel, get(), 0, duration.count());
401 }
402 }
403
404 void fade_out(const ms_type duration) noexcept(noexcept(duration.count())) // NOLINT
405 {
406 assert(duration.count() > 0);
407 if (is_playing()) {
408 Mix_FadeOutChannel(mChannel, duration.count());
409 }
410 }
411
412 [[nodiscard]] auto is_fading() const noexcept -> bool
413 {
414 return is_playing() && Mix_FadingChannel(mChannel);
415 }

424 }
425 }
