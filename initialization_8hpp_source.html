114class mix final
115{
116 public:
117 CENTURION_NODISCARD_CTOR explicit mix(const mix_cfg& cfg = {})
118 {
119 if (!Mix_Init(cfg.flags)) {
120 throw mix_error{};
121 }
122
123 if (Mix_OpenAudio(cfg.frequency, cfg.format, cfg.channels, cfg.chunk_size) == -1) {
124 throw mix_error{};
125 }
126 }
127
128 ~mix() noexcept
129 {
130 Mix_CloseAudio();
131 Mix_Quit();
132 }
133};
