61enum class texture_access

63 non_lockable = SDL_TEXTUREACCESS_STATIC,
64 streaming = SDL_TEXTUREACCESS_STREAMING,
65 target = SDL_TEXTUREACCESS_TARGET
66};

71[[nodiscard]] constexpr auto to_string(const texture_access access) -> std::string_view
72{
73 switch (access) {
74 case texture_access::non_lockable:
75 return "non_lockable";
76
77 case texture_access::streaming:
78 return "streaming";
79
80 case texture_access::target:
81 return "target";
82
83 default:
84 throw exception{"Did not recognize texture access!"};
85 }
86}
