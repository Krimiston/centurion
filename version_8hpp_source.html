59#define CENTURION_MAKE_VERSION_NUMBER(X, Y, Z) (((X)*1'000) + ((Y)*100) + (Z))
60
61#define CENTURION_VERSION_NUMBER \
62 CENTURION_MAKE_VERSION_NUMBER(CENTURION_VERSION_MAJOR, \
63 CENTURION_VERSION_MINOR, \
64 CENTURION_VERSION_PATCH)
65
66#define CENTURION_VERSION_AT_LEAST(X, Y, Z) \
67 CENTURION_VERSION_NUMBER >= CENTURION_MAKE_VERSION_NUMBER(X, Y, Z)

69#define CENTURION_SDL_VERSION_IS(X, Y, Z) \
70 ((SDL_MAJOR_VERSION == (X)) && (SDL_MINOR_VERSION == (Y)) && (SDL_PATCHLEVEL == (Z)))
