405 }
406 else {
407 return SDL_RenderDrawLinesF(get(), first, isize(container)) == 0;
408 }
409 }
410 else {
411 return failure;

439 template <typename X>
440 void draw_circle(const basic_point<X>& position, const float radius) noexcept
441 {
442 using value_t = typename basic_point<X>::value_type;
443
444 auto error = -radius;
445 auto x = radius - 0.5f;
446 auto y = 0.5f;
447
448 const auto cx = static_cast<float>(position.x()) - 0.5f;
449 const auto cy = static_cast<float>(position.y()) - 0.5f;
450
451 while (x >= y) {
452 draw_point<value_t>({static_cast<value_t>(cx + x), static_cast<value_t>(cy + y)});
453 draw_point<value_t>({static_cast<value_t>(cx + y), static_cast<value_t>(cy + x)});
454
455 if (x != 0) {
456 draw_point<value_t>({static_cast<value_t>(cx - x), static_cast<value_t>(cy + y)});
457 draw_point<value_t>({static_cast<value_t>(cx + y), static_cast<value_t>(cy - x)});
458 }
459
460 if (y != 0) {
461 draw_point<value_t>({static_cast<value_t>(cx + x), static_cast<value_t>(cy - y)});
462 draw_point<value_t>({static_cast<value_t>(cx - y), static_cast<value_t>(cy + x)});
463 }
464
465 if (x != 0 && y != 0) {
466 draw_point<value_t>({static_cast<value_t>(cx - x), static_cast<value_t>(cy - y)});
467 draw_point<value_t>({static_cast<value_t>(cx - y), static_cast<value_t>(cy - x)});
468 }
469
470 error += y;
471 ++y;
472 error += y;
473
474 if (error >= 0) {
475 --x;
476 error -= x;
477 error -= x;
478 }
479 }
480 }

488 template <typename X>
489 void fill_circle(const basic_point<X>& center, const float radius)
490 {
491 const auto cx = static_cast<float>(center.x());
492 const auto cy = static_cast<float>(center.y());
493
494 for (auto dy = 1.0f; dy <= radius; dy += 1.0f) {
495 const auto dx = std::floor(std::sqrt((2.0f * radius * dy) - (dy * dy)));
496 draw_line<float>({cx - dx, cy + dy - radius}, {cx + dx, cy + dy - radius});
497 draw_line<float>({cx - dx, cy - dy + radius}, {cx + dx, cy - dy + radius});
498 }
499 }

618 template <typename X, typename Y, typename Z>
619 auto render(const basic_texture<X>& texture,
620 const irect& src,
621 const basic_rect<Y>& dst,
622 const double angle,
623 const basic_point<Z>& center,
624 const renderer_flip flip) noexcept -> result
625 {
626 static_assert(std::is_same_v<typename basic_rect<Y>::value_type,
627 typename basic_point<Z>::value_type>,
628 "Destination rectangle and center point must have the same "
629 "value types (int or float)!");
630
631 if constexpr (basic_rect<Y>::floating) {
632 return SDL_RenderCopyExF(get(),
633 texture.get(),
634 src.data(),
635 dst.data(),
636 angle,
637 center.data(),
638 static_cast<SDL_RendererFlip>(flip)) == 0;
639 }
640 else {
641 return SDL_RenderCopyEx(get(),
642 texture.get(),
643 src.data(),
644 dst.data(),
645 angle,
646 center.data(),
647 static_cast<SDL_RendererFlip>(flip)) == 0;
648 }
649 }

1162 [[nodiscard]] auto capture(const pixel_format format) const -> surface
1163 {
1164 surface image{output_size(), format};
1165
1166 if (!image.lock()) {
1167 throw sdl_error{};
1168 }
1169
1170 if (const auto res =
1171 SDL_RenderReadPixels(get(), nullptr, 0, image.pixel_data(), image.pitch());
1172 res == -1) {
1173 throw sdl_error{};
1174 }
1175
1176 image.unlock();
1177 return image;
1178 }
