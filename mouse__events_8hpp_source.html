43enum class mouse_wheel_direction : uint32
44{
45 normal = SDL_MOUSEWHEEL_NORMAL,
46 flipped = SDL_MOUSEWHEEL_FLIPPED
47};

52[[nodiscard]] constexpr auto to_string(const mouse_wheel_direction dir) -> std::string_view
53{
54 switch (dir) {
55 case mouse_wheel_direction::normal:
56 return "normal";
57
58 case mouse_wheel_direction::flipped:
59 return "flipped";
60
61 default:
62 throw exception{"Did not recognize mouse wheel direction!"};
63 }
64}

66inline auto operator<<(std::ostream& stream, const mouse_wheel_direction dir) -> std::ostream&
67{
68 return stream << to_string(dir);
69}
