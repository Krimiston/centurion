68enum class display_event_id
69{
70 none = SDL_DISPLAYEVENT_NONE,
71 orientation = SDL_DISPLAYEVENT_ORIENTATION,
72 connected = SDL_DISPLAYEVENT_CONNECTED,
73 disconnected = SDL_DISPLAYEVENT_DISCONNECTED,
74};
75
78
79[[nodiscard]] constexpr auto to_string(const display_event_id id) -> std::string_view
80{
81 switch (id) {
82 case display_event_id::none:
83 return "none";
84
85 case display_event_id::orientation:
86 return "orientation";
87
88 case display_event_id::connected:
89 return "connected";
90
91 case display_event_id::disconnected:
92 return "disconnected";
93
94 default:
95 throw exception{"Did not recognize display event ID!"};
96 }
97}
98
99inline auto operator<<(std::ostream& stream, const display_event_id id) -> std::ostream&
100{
101 return stream << to_string(id);
102}
