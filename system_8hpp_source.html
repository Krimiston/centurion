137[[nodiscard]] inline auto to_string(const platform_id id) -> std::string_view
138{
139 switch (id) {
140 case platform_id::unknown:
141 return "unknown";
142
143 case platform_id::windows:
144 return "windows";
145
146 case platform_id::macos:
147 return "macos";
148
149 case platform_id::linux_os:

151
152 case platform_id::ios:

155 case platform_id::android:

157
158 default:
159 throw exception{"Did not recognize platform!"};

258[[nodiscard]] inline auto current_platform() noexcept -> platform_id
259{
260 const auto name = platform_name();
261 if (name == "Windows") {
262 return platform_id::windows;
263 }
264 else if (name == "Mac OS X") {
265 return platform_id::macos;
266 }
267 else if (name == "Linux") {
268 return platform_id::linux_os;
269 }
270 else if (name == "iOS") {
271 return platform_id::ios;
272 }
273 else if (name == "Android") {
274 return platform_id::android;
275 }
276 else {
277 return platform_id::unknown;
278 }
279}
