40enum class joy_hat_position : uint8
41{
42 left_up = SDL_HAT_LEFTUP,
43 left = SDL_HAT_LEFT,
44 left_down = SDL_HAT_LEFTDOWN,
45 up = SDL_HAT_UP,
46 centered = SDL_HAT_CENTERED,
47 down = SDL_HAT_DOWN,
48 right_up = SDL_HAT_RIGHTUP,
49 right = SDL_HAT_RIGHT,
50 right_down = SDL_HAT_RIGHTDOWN
51};
52
53[[nodiscard]] constexpr auto to_string(const joy_hat_position position) -> std::string_view
54{
55 switch (position) {
56 case joy_hat_position::left_up:
57 return "left_up";
58
59 case joy_hat_position::left:
60 return "left";
61
62 case joy_hat_position::left_down:
63 return "left_down";
64
65 case joy_hat_position::up:
66 return "up";
67
68 case joy_hat_position::centered:
69 return "centered";
70
71 case joy_hat_position::down:
72 return "down";
73
74 case joy_hat_position::right_up:
75 return "right_up";
76
77 case joy_hat_position::right:
78 return "right";
79
80 case joy_hat_position::right_down:
81 return "right_down";
82
83 default:
84 throw exception{"Did not recognize joystick hat position!"};
85 }
86}
87
88inline auto operator<<(std::ostream& stream, const joy_hat_position position) -> std::ostream&
89{
90 return stream << to_string(position);
91}
