729 [[nodiscard]] constexpr auto get_union(const basic_rect<T>& fst,
730  const basic_rect<T>& snd) noexcept -> basic_rect<T>
731 {
732  const auto fstHasArea = fst.has_area();
733  const auto sndHasArea = snd.has_area();
734 
735  if (!fstHasArea && !sndHasArea) {
736  return {};
737  }
738  else if (!fstHasArea) {
739  return snd;
740  }
741  else if (!sndHasArea) {
742  return fst;
743  }
744 
745  const auto x = detail::min(fst.x(), snd.x());
746  const auto y = detail::min(fst.y(), snd.y());
747  const auto maxX = detail::max(fst.max_x(), snd.max_x());
748  const auto maxY = detail::max(fst.max_y(), snd.max_y());
749 
750  return {{x, y}, {maxX - x, maxY - y}};
751 }
