190 void update() noexcept
191 {
192 mPreviousMask = mCurrentMask;
193 mPreviousPosition = mCurrentPosition;
194
195 int mx{};
196 int my{};
197 mCurrentMask = SDL_GetMouseState(&mx, &my);
198
199 mCurrentPosition.set_x(mx);
200 mCurrentPosition.set_y(my);
201 }

203#if SDL_VERSION_ATLEAST(2, 0, 18)
204
205 template <typename T>
206 void update(const basic_renderer<T>& renderer) noexcept
207 {
208 mPreviousMask = mCurrentMask;
209 mPreviousPosition = mCurrentPosition;
210
211 int mx{};
212 int my{};
213 mCurrentMask = SDL_GetMouseState(&mx, &my);
214
215 const auto logicalSize = renderer.logical_size();
216 if (logicalSize.width != 0 && logicalSize.height != 0) {
217 mCurrentPosition = renderer.to_logical(mx, my);
218 }
219 else {
220 /* No logical size has been set for the renderer */
221 mCurrentPosition.set_x(mx);
222 mCurrentPosition.set_y(my);
223 }
224 }
225
226#endif // SDL_VERSION_ATLEAST(2, 0, 18)

303
304 [[nodiscard]] auto is_pressed(const uint32 mask) const noexcept -> bool
305 {
306 return mCurrentMask & mask;
307 }
308
309 [[nodiscard]] auto was_released(const uint32 mask) const noexcept -> bool
310 {
311 return !(mCurrentMask & mask) && mPreviousMask & mask;
312 }
