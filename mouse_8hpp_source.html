285 template <typename T>
286 void update(const basic_renderer<T>& renderer) noexcept
287 {
288 mPreviousMask = mCurrentMask;
289 mPreviousPosition = mCurrentPosition;
290
291 int mx{};
292 int my{};
293 mCurrentMask = SDL_GetMouseState(&mx, &my);
294
295 const auto logicalSize = renderer.logical_size();
296 if (logicalSize.width != 0 && logicalSize.height != 0) {
297 mCurrentPosition = renderer.to_logical(mx, my);
298 }
299 else {
300 /* No logical size has been set for the renderer */
301 mCurrentPosition.set_x(mx);
302 mCurrentPosition.set_y(my);
303 }
304 }

306#endif // SDL_VERSION_ATLEAST(2, 0, 18)
