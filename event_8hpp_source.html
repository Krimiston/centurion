491template <typename E>
492class event_sink final
493{
494 public:
495 using event_type = std::decay_t<E>;
496 using signature_type = void(const event_type&);
497 using function_type = std::function<signature_type>;
498
500 void reset() noexcept { mFunction = nullptr; }
501
503 template <typename T>
504 void to(T&& callable)
505 {
506 static_assert(std::is_invocable_v<T, const event_type&>,
507 "Callable must be invocable with subscribed event!");
508
509 mFunction = function_type{callable};
510 }
511
513 template <auto MemberFunc, typename Self>
514 void to(Self* self)
515 {
516 static_assert(std::is_member_function_pointer_v<decltype(MemberFunc)>);
517 static_assert(std::is_invocable_v<decltype(MemberFunc), Self*, const event_type&>,
518 "Member function must be invocable with subscribed event!");
519
520 to(std::bind(MemberFunc, self, std::placeholders::_1));
521 }
522
524 template <auto Function>
525 void to()
526 {
527 to(Function);
528 }

562template <typename... Events>
563class event_dispatcher final
564{
565 static_assert((!std::is_const_v<Events> && ...));
566 static_assert((!std::is_volatile_v<Events> && ...));
567 static_assert((!std::is_reference_v<Events> && ...));
568 static_assert((!std::is_pointer_v<Events> && ...));
569
570 using sink_tuple = std::tuple<event_sink<Events>...>;
571
573 template <typename Event>
574 [[nodiscard]] constexpr static auto index_of() -> std::size_t
575 {
576 using sink_type = event_sink<std::decay_t<Event>>;
577
578 constexpr auto index = detail::tuple_type_index_v<sink_type, sink_tuple>;
579 static_assert(index != -1, "Invalid event type!");
580
581 return index;
582 }
583
584 template <typename Event>
585 [[nodiscard]] auto get_sink() -> event_sink<Event>&
586 {
587 constexpr auto index = index_of<Event>();
588 return std::get<index>(mSinks);
589 }
590
591 template <typename Event>
592 [[nodiscard]] auto get_sink() const -> const event_sink<Event>&
593 {
594 constexpr auto index = index_of<Event>();
595 return std::get<index>(mSinks);
596 }
597
599 template <typename Event>
600 auto check_for() -> bool
601 {
602 if (const auto* event = mEvent.template try_get<Event>()) {
603 auto& function = get_sink<Event>().function();
604
605 if (function) {
606 function(*event);
607 }
608
609 return true;
610 }
611 else {
612 return false;
613 }
614 }
615
616 public:
618 void poll()
619 {
620 while (mEvent.poll()) {
621 (check_for<Events>() || ...); /* Use OR to exploit short-circuiting */
622 }
623 }
624
632 template <typename Event>
633 auto bind() -> event_sink<Event>&
634 {
635 static_assert((std::is_same_v<std::decay_t<Event>, Events> || ...),
636 "Cannot connect unsubscribed event! Make sure that the "
637 "event is provided as a class template parameter.");
638 return get_sink<Event>();
639 }
640
642 void reset() noexcept { (bind<Events>().reset(), ...); }
643
645 [[nodiscard]] auto active_count() const -> std::size_t
646 {
647 return (0u + ... + (get_sink<Events>().function() ? 1u : 0u));
648 }
649
651 [[nodiscard]] constexpr static auto size() noexcept -> std::size_t
652 {
653 return sizeof...(Events);
654 }
655
656 private:
657 cen::event_handler mEvent;
658 sink_tuple mSinks;
659};

661template <typename... E>
662[[nodiscard]] auto to_string(const event_dispatcher<E...>& dispatcher) -> std::string
663{
664#if CENTURION_HAS_FEATURE_FORMAT
665 return std::format("event_dispatcher(size: {}, #active: {})",
666 dispatcher.size(),
667 dispatcher.active_count());
668#else
669 return "event_dispatcher(size: " + std::to_string(dispatcher.size()) +
670 ", #active: " + std::to_string(dispatcher.active_count()) + ")";
671#endif // CENTURION_HAS_FEATURE_FORMAT
672}

676 -> std::ostream&

678 return stream << to_string(dispatcher);
679}
