93enum class font_hint
94{
95 normal = TTF_HINTING_NORMAL,
96 light = TTF_HINTING_LIGHT,
97
98#if SDL_TTF_VERSION_ATLEAST(2, 0, 18)
99 light_subpixel = TTF_HINTING_LIGHT_SUBPIXEL,
100#endif // SDL_TTF_VERSION_ATLEAST(2, 0, 18)
101
102 mono = TTF_HINTING_MONO,
103 none = TTF_HINTING_NONE
104};
105
106[[nodiscard]] constexpr auto to_string(const font_hint hint) -> std::string_view
107{
108 switch (hint) {
109 case font_hint::normal:
110 return "normal";
111
112 case font_hint::light:
113 return "light";
114
115#if SDL_TTF_VERSION_ATLEAST(2, 0, 18)
116
117 case font_hint::light_subpixel:
118 return "light_subpixel";
119
120#endif // SDL_TTF_VERSION_ATLEAST(2, 0, 18)
121
122 case font_hint::mono:
123 return "mono";
124
125 case font_hint::none:
126 return "none";
127
128 default:
129 throw exception{"Did not recognize font hint!"};
130 }
131}
132
133inline auto operator<<(std::ostream& stream, const font_hint hint) -> std::ostream&
134{
135 return stream << to_string(hint);
136}

768 template <typename T>
769 auto render_glyph(basic_renderer<T>& renderer, const unicode_t glyph, const ipoint& position)
770 -> int
771 {
772 if (const auto* data = find_glyph(glyph)) {
773 const auto& [texture, metrics] = *data;
774 const auto outline = mFont.outline();
775
776 /* SDL_ttf handles the y-axis alignment */
777 const auto x = position.x() + metrics.min_x - outline;
778 const auto y = position.y() - outline;
779
780 renderer.render(texture, ipoint{x, y});
781
782 return x + metrics.advance;
783 }
784 else {
785 return position.x();
786 }
787 }

801 template <typename T, typename String>
802 void render_text(basic_renderer<T>& renderer, const String& str, ipoint position)
803 {
804 const auto originalX = position.x();
805 const auto lineSkip = mFont.line_skip();
806
807 for (const unicode_t glyph : str) {
808 if (glyph == '\n') {
809 position.set_x(originalX);
810 position.set_y(position.y() + lineSkip);
811 }
812 else {
813 const auto x = render_glyph(renderer, glyph, position);
814 position.set_x(x);
815 }
816 }
817 }

835 template <typename T>
