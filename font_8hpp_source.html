93enum class font_hint
94{
95 normal = TTF_HINTING_NORMAL,
96 light = TTF_HINTING_LIGHT,
97
98#if SDL_TTF_VERSION_ATLEAST(2, 0, 18)
99 light_subpixel = TTF_HINTING_LIGHT_SUBPIXEL,
100#endif // SDL_TTF_VERSION_ATLEAST(2, 0, 18)
101
102 mono = TTF_HINTING_MONO,
103 none = TTF_HINTING_NONE
104};
105
106[[nodiscard]] constexpr auto to_string(const font_hint hint) -> std::string_view
107{
108 switch (hint) {
109 case font_hint::normal:
110 return "normal";
111
112 case font_hint::light:
113 return "light";
114
115#if SDL_TTF_VERSION_ATLEAST(2, 0, 18)
116
117 case font_hint::light_subpixel:
118 return "light_subpixel";
119
120#endif // SDL_TTF_VERSION_ATLEAST(2, 0, 18)
121
122 case font_hint::mono:
123 return "mono";
124
125 case font_hint::none:
126 return "none";
127
128 default:
129 throw exception{"Did not recognize font hint!"};
130 }
131}
132
133inline auto operator<<(std::ostream& stream, const font_hint hint) -> std::ostream&
134{
135 return stream << to_string(hint);
136}

815 auto render_glyph(basic_renderer<T>& renderer, const unicode_t glyph, const ipoint& position)
816 -> int
817 {
818 if (const auto* data = find_glyph(glyph)) {
819 const auto& [texture, metrics] = *data;
820 const auto outline = mFont.outline();
821
822 /* SDL_ttf handles the y-axis alignment */
823 const auto x = position.x() + metrics.min_x - outline;
824 const auto y = position.y() - outline;
825
826 renderer.render(texture, ipoint{x, y});
827
828 return x + metrics.advance;
829 }
830 else {
831 return position.x();
832 }
833 }

847 template <typename T, typename String>
848 void render_text(basic_renderer<T>& renderer, const String& str, ipoint position)
849 {
850 const auto originalX = position.x();
851 const auto lineSkip = mFont.line_skip();
852
853 for (const unicode_t glyph : str) {
854 if (glyph == '\n') {
855 position.set_x(originalX);
856 position.set_y(position.y() + lineSkip);
857 }
858 else {
859 const auto x = render_glyph(renderer, glyph, position);
860 position.set_x(x);
861 }
862 }
863 }
