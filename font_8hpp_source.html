122enum class font_hint
123{
124 normal = TTF_HINTING_NORMAL,
125 light = TTF_HINTING_LIGHT,
126
127#if SDL_TTF_VERSION_ATLEAST(2, 0, 18)
128 light_subpixel = TTF_HINTING_LIGHT_SUBPIXEL,
129#endif // SDL_TTF_VERSION_ATLEAST(2, 0, 18)
130
131 mono = TTF_HINTING_MONO,
132 none = TTF_HINTING_NONE
133};

137
138[[nodiscard]] constexpr auto to_string(const font_hint hint) -> std::string_view
139{
140 switch (hint) {
141 case font_hint::normal:
142 return "normal";
143
144 case font_hint::light:
145 return "light";
146
147#if SDL_TTF_VERSION_ATLEAST(2, 0, 18)
148
149 case font_hint::light_subpixel:
150 return "light_subpixel";
151
152#endif // SDL_TTF_VERSION_ATLEAST(2, 0, 18)
153
154 case font_hint::mono:
155 return "mono";
156
157 case font_hint::none:
158 return "none";
159
160 default:
161 throw exception{"Did not recognize font hint!"};
162 }
163}
164
165inline auto operator<<(std::ostream& stream, const font_hint hint) -> std::ostream&
166{

1147 template <typename T>
1148 auto render_glyph(basic_renderer<T>& renderer, const unicode_t glyph, const ipoint& position)
1149 -> int
1150 {
1151 if (const auto* data = find_glyph(glyph)) {
1152 const auto& [texture, metrics] = *data;
1153 const auto outline = mFont.outline();
1154
1155 /* SDL_ttf handles the y-axis alignment */
1156 const auto x = position.x() + metrics.min_x - outline;
1157 const auto y = position.y() - outline;
1158
1159 renderer.render(texture, ipoint{x, y});
1160
1161 return x + metrics.advance;
1162 }
1163 else {
1164 return position.x();
1165 }
