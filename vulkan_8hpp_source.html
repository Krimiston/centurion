129inline auto required_extensions() -> std::optional<std::vector<const char*>>
130{
131 uint count{};
132 if (!SDL_Vulkan_GetInstanceExtensions(nullptr, &count, nullptr)) {
133 return std::nullopt;
134 }
135
136 std::vector<const char*> names(count);
137 if (!SDL_Vulkan_GetInstanceExtensions(nullptr, &count, names.data())) {
138 return std::nullopt;
139 }
140
141 return names;
142}
