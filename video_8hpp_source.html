328[[nodiscard]] inline auto compose_blend_mode(const blend_task& color,
329 const blend_task& alpha) noexcept -> blend_mode
330{
331 const auto res = SDL_ComposeCustomBlendMode(static_cast<SDL_BlendFactor>(color.src),
332 static_cast<SDL_BlendFactor>(color.dst),
333 static_cast<SDL_BlendOperation>(color.op),
334 static_cast<SDL_BlendFactor>(alpha.src),
335 static_cast<SDL_BlendFactor>(alpha.dst),
336 static_cast<SDL_BlendOperation>(alpha.op));
337 return static_cast<blend_mode>(res);
338}
