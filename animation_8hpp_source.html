65 [[nodiscard]] static auto load(const char* file) -> animation
66 {

81 [[nodiscard]] auto at(const usize index) -> surface_handle
82 {
83 if (index < count()) {
84 return surface_handle{mAnim->frames[index]};
85 }
86 else {
87 throw exception{"Invalid animation frame index!"};
88 }
89 }
90
91 [[nodiscard]] auto operator[](const usize index) -> surface_handle { return at(index); }
92
93 [[nodiscard]] auto delay(const usize index) const -> int
94 {
95 if (index < count()) {
96 return mAnim->delays[index];
97 }
98 else {
99 throw exception{"Invalid animation frame index!"};
100 }
101 }
102
103 [[nodiscard]] auto count() const noexcept -> int { return mAnim->count; }
104
105 [[nodiscard]] auto width() const noexcept -> int { return mAnim->w; }
106 [[nodiscard]] auto height() const noexcept -> int { return mAnim->h; }
107 [[nodiscard]] auto size() const noexcept -> iarea { return {width(), height()}; }
108
109 [[nodiscard]] auto get() const noexcept -> IMG_Animation* { return mAnim.get(); }

111 private:
