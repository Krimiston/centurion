184template <typename T>
185using managed_ptr = std::unique_ptr<T, deleter<T>>;

187class simd_block final
188{
189 public:
190 using size_type = std::size_t;
191
192 explicit simd_block(const size_type size) noexcept : mData{SDL_SIMDAlloc(size)} {}
193
194#if SDL_VERSION_ATLEAST(2, 0, 14)
195
196 void reallocate(const size_type size) noexcept
197 {
198 /* We temporarily release the ownership of the pointer in order to avoid a double
199 delete, since the reallocation will free the previously allocated memory. */
200 auto* ptr = mData.release();
201 mData.reset(SDL_SIMDRealloc(ptr, size));
202 }
203
204#endif // SDL_VERSION_ATLEAST(2, 0, 14)
205
206 [[nodiscard]] auto data() noexcept -> void* { return mData.get(); }
207
208 [[nodiscard]] auto data() const noexcept -> const void* { return mData.get(); }
209
211 explicit operator bool() const noexcept { return mData != nullptr; }
212
213 private:
214 struct simd_deleter final
215 {
216 void operator()(void* ptr) noexcept { SDL_SIMDFree(ptr); }
217 };
218 std::unique_ptr<void, simd_deleter> mData;
219};
