64enum class log_priority
65{
66 verbose = SDL_LOG_PRIORITY_VERBOSE,
67 debug = SDL_LOG_PRIORITY_DEBUG,
68 info = SDL_LOG_PRIORITY_INFO,
69 warn = SDL_LOG_PRIORITY_WARN,
70 error = SDL_LOG_PRIORITY_ERROR,
71 critical = SDL_LOG_PRIORITY_CRITICAL,
72};

77[[nodiscard]] constexpr auto to_string(const log_priority priority) -> std::string_view
78{
79 switch (priority) {

83 case log_priority::debug:

132[[nodiscard]] constexpr auto is_custom(const log_category category) noexcept -> bool
133{
134 return to_underlying(category) >= SDL_LOG_CATEGORY_CUSTOM;

202inline void set_priority(const log_priority priority) noexcept
203{
204 const auto value = static_cast<SDL_LogPriority>(priority);

242template <typename... Args>
243void log(const log_priority priority,
244 const log_category category,
245 const char* fmt,
246 Args&&... args) noexcept
247{
248 assert(fmt);
249 SDL_LogMessage(static_cast<SDL_LogCategory>(category),
250 static_cast<SDL_LogPriority>(priority),
251 fmt,
252 std::forward<Args>(args)...);
253}
