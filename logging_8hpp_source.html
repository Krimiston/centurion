49enum class log_priority
50{
51 verbose = SDL_LOG_PRIORITY_VERBOSE,
52 debug = SDL_LOG_PRIORITY_DEBUG,
53 info = SDL_LOG_PRIORITY_INFO,
54 warn = SDL_LOG_PRIORITY_WARN,
55 error = SDL_LOG_PRIORITY_ERROR,
56 critical = SDL_LOG_PRIORITY_CRITICAL,
57};

59[[nodiscard]] constexpr auto to_string(const log_priority priority) -> std::string_view
60{
61 switch (priority) {
62 case log_priority::verbose:
63 return "verbose";
64
65 case log_priority::debug:
66 return "debug";
67
68 case log_priority::info:
69 return "info";
70
71 case log_priority::warn:
72 return "warn";
73
74 case log_priority::error:
75 return "error";
76
77 case log_priority::critical:
78 return "critical";
79
80 default:
81 throw exception{"Did not recognize log priority!"};
82 }
83}

98 render = SDL_LOG_CATEGORY_RENDER,

104[[nodiscard]] constexpr auto is_custom(const log_category category) noexcept -> bool
105{
106 return to_underlying(category) >= SDL_LOG_CATEGORY_CUSTOM;
107}

158inline void set_priority(const log_priority priority) noexcept
159{
160 const auto value = static_cast<SDL_LogPriority>(priority);
161 SDL_LogSetAllPriority(value);
162 SDL_LogSetPriority(SDL_LOG_CATEGORY_TEST, value); /* Apparently not set by SDL */
163}

165inline void set_priority(const log_category category, const log_priority priority) noexcept
166{
167 SDL_LogSetPriority(to_underlying(category), static_cast<SDL_LogPriority>(priority));
168}

180template <typename... Args>
181void log(const log_priority priority,
182 const log_category category,
183 const char* fmt,
184 Args&&... args) noexcept
185{
186 assert(fmt);
187 SDL_LogMessage(static_cast<SDL_LogCategory>(category),
188 static_cast<SDL_LogPriority>(priority),
189 fmt,
190 std::forward<Args>(args)...);
191}
