422 explicit basic_controller(maybe_owner<SDL_GameController*> controller) noexcept(detail::is_handle<T>)

512 [[nodiscard]] static auto count() noexcept -> int
513 {
514 const auto joysticks = SDL_NumJoysticks();

669 [[nodiscard]] auto touchpad_finger_state(const int touchpad, const int finger) const noexcept
670 -> std::optional<controller_finger_state>
671 {
672 controller_finger_state result;
673 uint8 state{};
674
675 const auto res = SDL_GameControllerGetTouchpadFinger(mController,
676 touchpad,
677 finger,
678 &state,
679 &result.x,
680 &result.y,
681 &result.pressure);
682 result.state = static_cast<button_state>(state);
683
684 if (res != -1) {
685 return result;
686 }
687 else {
688 return std::nullopt;
689 }
