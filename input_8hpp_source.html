37enum class button_state : uint8
38{
39 released = SDL_RELEASED,
40 pressed = SDL_PRESSED
41};
42
43[[nodiscard]] constexpr auto to_string(const button_state state) -> std::string_view
44{
45 switch (state) {
46 case button_state::released:
47 return "released";
48
49 case button_state::pressed:
50 return "pressed";
51
52 default:
53 throw exception{"Did not recognize button state!"};
54 }
55}
56
57inline auto operator<<(std::ostream& stream, const button_state state) -> std::ostream&
58{
59 return stream << to_string(state);
