37using owner_tag = std::true_type;
38using handle_tag = std::false_type;

40template <typename T>
41inline constexpr bool is_owner = std::is_same_v<T, owner_tag>;

43template <typename T>
44inline constexpr bool is_handle = std::is_same_v<T, handle_tag>;

52template <typename B, typename Type>
53class pointer final
54{
55 public:
56 using smart_ptr = managed_ptr<Type>;
57 using raw_ptr = Type*;
58 using pointer_type = std::conditional_t<B::value, smart_ptr, raw_ptr>;
59
60 pointer() noexcept = default;
61
62 explicit pointer(Type* ptr) noexcept : mPtr{ptr} {}
63
64 template <typename BB = B, enable_for_owner<BB> = 0>
65 void reset(Type* ptr) noexcept
66 {
67 mPtr.reset(ptr);
68 }
69
70 auto operator->() noexcept -> Type* { return get(); }
71
72 auto operator->() const noexcept -> const Type* { return get(); }
73
74 auto operator*() noexcept -> Type&
75 {
76 assert(mPtr);
77 return *mPtr;
78 }
79
80 auto operator*() const noexcept -> const Type&
81 {
82 assert(mPtr);
83 return *mPtr;
84 }
85
86 explicit operator bool() const noexcept { return mPtr != nullptr; }
87
88 /*implicit*/ operator Type*() const noexcept { return get(); }
89
90 template <typename BB = B, enable_for_owner<BB> = 0>
91 [[nodiscard]] auto release() noexcept -> Type*
92 {
93 return mPtr.release();
94 }
95
96 [[nodiscard]] auto get() const noexcept -> Type*
97 {
98 if constexpr (B::value) {
99 return mPtr.get();
100 }
101 else {
102 return mPtr;
103 }
104 }
105
106 private:
107 pointer_type mPtr{};
108};
