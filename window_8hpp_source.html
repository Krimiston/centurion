134 explicit basic_window(maybe_owner<SDL_Window*> window) noexcept(detail::is_handle<T>)
135 : mWindow{window}
136 {
137 if constexpr (detail::is_owner<T>) {
138 if (!mWindow) {
139 throw exception{"Cannot create window from null pointer!"};
140 }
141 }

157 template <typename TT = T, detail::enable_for_owner<TT> = 0>
158 explicit basic_window(const char* title,
159 const iarea size = default_size(),
160 const uint32 flags = default_flags())
161 {
162 assert(title);
163
164 if (size.width < 1) {
165 throw exception{"Bad window width!"};
166 }
167 else if (size.height < 1) {
168 throw exception{"Bad window height!"};
169 }
170
171 mWindow.reset(SDL_CreateWindow(title,
172 SDL_WINDOWPOS_CENTERED,
173 SDL_WINDOWPOS_CENTERED,
174 size.width,
175 size.height,
176 flags));
177 if (!mWindow) {
178 throw sdl_error{};
179 }
180 }

185 const iarea size = default_size(),
186 const uint32 flags = default_flags())
187 : basic_window{title.c_str(), size, flags}
188 {}
