119 explicit basic_window(maybe_owner<SDL_Window*> window) noexcept(detail::is_handle<T>)
120 : mWindow{window}
121 {
122 if constexpr (detail::is_owner<T>) {
123 if (!mWindow) {
124 throw exception{"Cannot create window from null pointer!"};
125 }
126 }

129 template <typename TT = T, detail::enable_for_owner<TT> = 0>
130 explicit basic_window(const char* title,
131 const iarea size = default_size(),
132 const uint32 flags = default_flags())
133 {
134 assert(title);
135
136 if (size.width < 1) {
137 throw exception{"Bad window width!"};
138 }
139 else if (size.height < 1) {
140 throw exception{"Bad window height!"};
141 }
142
143 mWindow.reset(SDL_CreateWindow(title,
144 SDL_WINDOWPOS_CENTERED,
145 SDL_WINDOWPOS_CENTERED,
146 size.width,
147 size.height,
148 flags));
149 if (!mWindow) {
150 throw sdl_error{};
151 }
152 }

159 {}
