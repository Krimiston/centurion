256template <typename T>
257class basic_gl_context final
258{
259 public:
260 explicit basic_gl_context(maybe_owner<SDL_GLContext> context) noexcept(detail::is_handle<T>)
261 : mContext{context}
262 {
263 if constexpr (detail::is_owner<T>) {
264 if (!mContext) {
265 throw exception{"Can't create OpenGL context from null pointer!"};
266 }
267 }
268 }
269
270 template <typename U>
271 explicit basic_gl_context(basic_window<U>& window) noexcept(detail::is_handle<T>)
272 : mContext{SDL_GL_CreateContext(window.get())}
273 {
274 if constexpr (detail::is_owner<T>) {
275 if (!mContext) {
276 throw sdl_error{};
277 }
278 }
279 }
280
281 template <typename U>
282 auto make_current(basic_window<U>& window) -> result
283 {
284 assert(window.is_opengl());
285 return SDL_GL_MakeCurrent(window.get(), mContext.get()) == 0;
286 }
287
288 [[nodiscard]] auto get() const noexcept -> SDL_GLContext { return mContext.get(); }
289
290 private:
291 struct Deleter final
292 {
293 void operator()(SDL_GLContext context) noexcept { SDL_GL_DeleteContext(context); }
294 };
295
296 std::unique_ptr<void, Deleter> mContext;
297};
