274template <typename T>
275class basic_gl_context final
276{
277 public:
278 explicit basic_gl_context(maybe_owner<SDL_GLContext> context) noexcept(detail::is_handle<T>)
279 : mContext{context}
280 {
281 if constexpr (detail::is_owner<T>) {
282 if (!mContext) {
283 throw exception{"Can't create OpenGL context from null pointer!"};
284 }
285 }
286 }
287
288 template <typename U>
289 explicit basic_gl_context(basic_window<U>& window) noexcept(detail::is_handle<T>)
290 : mContext{SDL_GL_CreateContext(window.get())}
291 {
292 if constexpr (detail::is_owner<T>) {
293 if (!mContext) {
294 throw sdl_error{};
295 }
296 }
297 }
298
299 template <typename U>
300 auto make_current(basic_window<U>& window) -> result
301 {
302 assert(window.is_opengl());
303 return SDL_GL_MakeCurrent(window.get(), mContext.get()) == 0;
304 }
305
306 [[nodiscard]] auto get() const noexcept -> SDL_GLContext { return mContext.get(); }
307
308 private:
309 struct Deleter final
310 {
311 void operator()(SDL_GLContext context) noexcept { SDL_GL_DeleteContext(context); }
312 };
313
314 std::unique_ptr<void, Deleter> mContext;
315};
