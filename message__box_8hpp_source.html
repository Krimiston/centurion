484  auto show(SDL_Window* parent) -> std::optional<button_id>
485  {
486  SDL_MessageBoxData data{};
487 
488  data.window = parent;
489  data.title = m_title.c_str();
490  data.message = m_message.c_str();
491  data.flags = to_flags(m_type, m_buttonOrder);
492  data.colorScheme = m_colorScheme ? m_colorScheme->get() : nullptr;
493 
494 #if CENTURION_HAS_FEATURE_MEMORY_RESOURCE
495  // Realistically 1-3 buttons, stack buffer for 8 buttons, just in case.
496  detail::stack_resource<8 * sizeof(SDL_MessageBoxButtonData)> resource;
497  std::pmr::vector<SDL_MessageBoxButtonData> buttonData{resource.get()};
498 #else
499  std::vector<SDL_MessageBoxButtonData> buttonData;
500  buttonData.reserve(8);
501 #endif // CENTURION_HAS_STD_MEMORY_RESOURCE
502 
503  if (m_buttons.empty()) {
504  add_button(0, "OK", default_button::return_key);
505  }
506 
507  for (const auto& button : m_buttons) {
508  buttonData.emplace_back(button.convert());
509  }
510 
511  data.buttons = buttonData.data();
512  data.numbuttons = isize(buttonData);
513 
514  button_id button{-1};
515  if (SDL_ShowMessageBox(&data, &button) == -1) {
516  throw sdl_error{};
517  }
518 
519  if (button != -1) {
520  return button;
521  }
522  else {
523  return std::nullopt;
524  }
525  }
