559 static void show(SDL_Window* parent,
560 const char* title,
561 const char* message,
562 const message_box_type type,
563 const message_box_button_order order)
564 {
565 assert(title);
566 assert(message);
567
568 const auto res = SDL_ShowSimpleMessageBox(to_flags(type, order), title, message, parent);
569 if (res == -1) {
570 throw sdl_error{};
571 }
572 }
573
574 auto show(SDL_Window* parent) -> std::optional<button_id>
575 {
576 SDL_MessageBoxData data{};
577
578 data.window = parent;
579 data.title = mTitle.c_str();
580 data.message = mMessage.c_str();
581 data.flags = to_flags(mType, mButtonOrder);
582 data.colorScheme = mColorScheme ? mColorScheme->data() : nullptr;
583
584 std::vector<SDL_MessageBoxButtonData> buttonData;
585 buttonData.reserve(8);
586
587 if (mButtons.empty()) {
588 add_button(0, "OK", button_flags::return_key_default);
589 }
590
591 for (const auto& button : mButtons) {
592 buttonData.emplace_back(button.convert());
593 }
594
595 data.buttons = buttonData.data();
596 data.numbuttons = isize(buttonData);
597
598 button_id button{-1};
599 if (SDL_ShowMessageBox(&data, &button) == -1) {
600 throw sdl_error{};
601 }
602
603 if (button != -1) {
604 return button;
605 }
606 else {
607 return std::nullopt;
608 }
609 }
