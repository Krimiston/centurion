336 static void show(SDL_Window* parent,
337 const char* title,
338 const char* message,
339 const message_box_type type,
340 const message_box_button_order order)
341 {
342 assert(title);
343 assert(message);
344
345 const auto res = SDL_ShowSimpleMessageBox(to_flags(type, order), title, message, parent);
346 if (res == -1) {
347 throw sdl_error{};
348 }
349 }
350
351 auto show(SDL_Window* parent) -> std::optional<button_id>
352 {
353 SDL_MessageBoxData data{};
354
355 data.window = parent;
356 data.title = mTitle.c_str();
357 data.message = mMessage.c_str();
358 data.flags = to_flags(mType, mButtonOrder);
359 data.colorScheme = mColorScheme ? mColorScheme->data() : nullptr;
360
361 std::vector<SDL_MessageBoxButtonData> buttonData;
362 buttonData.reserve(8);
363
364 if (mButtons.empty()) {
365 add_button(0, "OK", button_flags::return_key_default);
366 }
367
368 for (const auto& button : mButtons) {
369 buttonData.emplace_back(button.convert());
370 }
371
372 data.buttons = buttonData.data();
373 data.numbuttons = isize(buttonData);
374
375 button_id button{-1};
376 if (SDL_ShowMessageBox(&data, &button) == -1) {
377 throw sdl_error{};
378 }
379
380 if (button != -1) {
381 return button;
382 }
383 else {
384 return std::nullopt;
385 }
386 }
