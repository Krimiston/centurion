339 template <is_stateless_callable Callable>
340 [[nodiscard]] static auto init([[maybe_unused]] Callable&& task, const char* name = "thread")
341 -> thread
342 {
343 assert(name);
344
345 constexpr bool nothrow = noexcept(Callable{}());
346
347 const auto wrapper = [](void* /*data*/) noexcept(nothrow) -> int {
348 Callable callable;
349 if constexpr (std::convertible_to<std::invoke_result_t<Callable>, int>) {
350 return callable();
351 }
352 else {
353 callable();
354 return 0;
355 }
356 };
357
358 return thread{wrapper, name};
359 }
360
361 template <typename T = void, is_stateless_callable<T*> Callable>
362 [[nodiscard]] static auto init([[maybe_unused]] Callable&& task,
363 T* userData = nullptr,
364 const char* name = "thread") -> thread
365 {
366 assert(name);
367
368 constexpr bool nothrow = noexcept(Callable{}(std::declval<T*>()));
369
370 const auto wrapper = [](void* erased) noexcept(nothrow) -> int {
371 auto* ptr = static_cast<T*>(erased);
372
373 Callable callable;
374 if constexpr (std::convertible_to<std::invoke_result_t<Callable, T*>, int>) {
375 return callable(ptr);
376 }
377 else {
378 callable(ptr);
379 return 0;
380 }
381 };
382
383 return thread{wrapper, name, userData};
384 }

399 auto join() noexcept -> int
400 {
401 if (mJoined || mDetached) {
402 return 0;
403 }
404
405 int status{};
406 SDL_WaitThread(mThread, &status);
407
408 mJoined = true;
409 assert(mDetached != mJoined);
410
411 return status;
412 }
413
415 void detach() noexcept
416 {
417 if (mJoined || mDetached) {
418 return;
419 }
420
421 SDL_DetachThread(mThread);
422
423 mDetached = true;
424 assert(mDetached != mJoined);
425 }
