515 template <is_stateless_callable Callable>
516 [[nodiscard]] static auto init([[maybe_unused]] Callable&& task, const char* name = "thread")
517 -> thread
518 {
519 assert(name);
520
521 constexpr bool nothrow = noexcept(Callable{}());
522
523 const auto wrapper = [](void* /*data*/) noexcept(nothrow) -> int {
524 Callable callable;
525 if constexpr (std::convertible_to<std::invoke_result_t<Callable>, int>) {
526 return callable();
527 }
528 else {
529 callable();
530 return 0;
531 }
532 };
533
534 return thread{wrapper, name};
535 }
536
554 template <typename T = void, is_stateless_callable<T*> Callable>
555 [[nodiscard]] static auto init([[maybe_unused]] Callable&& task,
556 T* userData = nullptr,
557 const char* name = "thread") -> thread
558 {
559 assert(name);
560
561 constexpr bool nothrow = noexcept(Callable{}(std::declval<T*>()));
562
563 const auto wrapper = [](void* erased) noexcept(nothrow) -> int {
564 auto* ptr = static_cast<T*>(erased);
565
566 Callable callable;
567 if constexpr (std::convertible_to<std::invoke_result_t<Callable, T*>, int>) {
568 return callable(ptr);
569 }
570 else {
571 callable(ptr);
572 return 0;
573 }
574 };
575
576 return thread{wrapper, name, userData};
577 }

618 auto join() noexcept -> int
619 {
620 if (mJoined || mDetached) {
621 return 0;
622 }
623
624 int status{};
625 SDL_WaitThread(mThread, &status);
626
627 mJoined = true;
628 assert(mDetached != mJoined);
629
630 return status;
631 }
632
638 void detach() noexcept
639 {
640 if (mJoined || mDetached) {
641 return;
642 }
643
644 SDL_DetachThread(mThread);
645
646 mDetached = true;
647 assert(mDetached != mJoined);
648 }
