100[[nodiscard]] constexpr auto to_string(const touch_device_type type) -> std::string_view
101{
102 switch (type) {
103 case touch_device_type::invalid:
104 return "invalid";
105
106 case touch_device_type::direct:
107 return "direct";
108
109 case touch_device_type::indirect_absolute:
