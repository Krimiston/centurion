48enum class touch_device_type
49{
50 invalid = SDL_TOUCH_DEVICE_INVALID,
51 direct = SDL_TOUCH_DEVICE_DIRECT,
52 indirect_absolute = SDL_TOUCH_DEVICE_INDIRECT_ABSOLUTE,
53 indirect_relative = SDL_TOUCH_DEVICE_INDIRECT_RELATIVE
54};
55
56[[nodiscard]] constexpr auto to_string(const touch_device_type type) -> std::string_view
57{
58 switch (type) {
59 case touch_device_type::invalid:
60 return "invalid";
61
62 case touch_device_type::direct:
63 return "direct";
64
65 case touch_device_type::indirect_absolute:
66 return "indirect_absolute";
67
68 case touch_device_type::indirect_relative:
69 return "indirect_relative";
70
71 default:
72 throw exception{"Did not recognize touch device type!"};
73 }
74}
75
76inline auto operator<<(std::ostream& stream, const touch_device_type type) -> std::ostream&
77{
78 return stream << to_string(type);
79}

87 [[nodiscard]] static auto find(const touch_id id, const int index) noexcept
88 -> std::optional<finger>
89 {
90 if (const auto* data = SDL_GetTouchFinger(id, index)) {
91 return finger{*data};
92 }
93 else {
94 return std::nullopt;
95 }
96 }
