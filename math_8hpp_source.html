282template <typename T>
283[[nodiscard]] auto distance(const basic_point<T>& from, const basic_point<T>& to) noexcept ->
284 typename basic_point<T>::value_type
285{
286 const auto distance =
287 std::sqrt(std::pow(from.x() - to.x(), T{2}) + std::pow(from.y() - to.y(), T{2}));
288 if constexpr (basic_point<T>::integral) {
289 return static_cast<int>(std::round(distance));
290 }
291 else {
292 return distance;
293 }
294}

588template <typename T>
589[[nodiscard]] constexpr auto get_union(const basic_rect<T>& a, const basic_rect<T>& b) noexcept
590 -> basic_rect<T>
591{
592 if constexpr (detail::sdl_version_at_least(2, 0, 22)) {
593 if constexpr (basic_rect<T>::floating) {
594 cen::frect res;
595 SDL_UnionFRect(a.data(), b.data(), res.data());
596 return res;
597 }
598 else {
599 cen::irect res;
600 SDL_UnionRect(a.data(), b.data(), res.data());
601 return res;
602 }
603 }
604 else {
605 const auto aHasArea = a.has_area();
606 const auto bHasArea = b.has_area();
607
608 if (!aHasArea && !bHasArea) {
609 return {};
610 }
611 else if (!aHasArea) {
612 return b;
613 }
614 else if (!bHasArea) {
615 return a;
616 }
617
618 const auto x = detail::min(a.x(), b.x());
619 const auto y = detail::min(a.y(), b.y());
620 const auto maxX = detail::max(a.max_x(), b.max_x());
621 const auto maxY = detail::max(a.max_y(), b.max_y());
622
623 return {{x, y}, {maxX - x, maxY - y}};
624 }
625}
