36template <typename T, usize Size>
37constexpr void assign(const std::array<T, Size>& array, bounded_array_ref<T, Size> out)
38{
39 usize index = 0;
40 for (auto&& value : array) {
41 out[index] = value;
42 ++index;
43 }
44}

46template <typename T, usize Size>
47[[nodiscard]] constexpr auto to_array(bounded_array_ref<const T, Size> data)
48 -> std::array<T, Size>
49{
50#if CENTURION_HAS_FEATURE_TO_ARRAY
51 return std::to_array(data);
52#else
53 std::array<T, Size> array; // NOLINT
54
55 for (usize i = 0; i < Size; ++i) {
56 array[i] = data[i];
57 }
58
59 return array;
60#endif // CENTURION_HAS_FEATURE_TO_ARRAY
61}
