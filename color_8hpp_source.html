124 [[nodiscard]] static auto from_hsv(float hue, float saturation, float value) -> color
125 {
126 hue = detail::clamp(hue, 0.0f, 360.0f);
127 saturation = detail::clamp(saturation, 0.0f, 100.0f);
128 value = detail::clamp(value, 0.0f, 100.0f);
129
130 const auto v = (value / 100.0f);
131 const auto chroma = v * (saturation / 100.0f);
132 const auto hp = hue / 60.0f;
133
134 const auto x = chroma * (1.0f - std::abs(std::fmod(hp, 2.0f) - 1.0f));
135
136 float red{};
137 float green{};
138 float blue{};
139
140 if (0 <= hp && hp <= 1) {
141 red = chroma;
142 green = x;
143 blue = 0;
144 }
145 else if (1 < hp && hp <= 2) {
146 red = x;
147 green = chroma;
148 blue = 0;
149 }
150 else if (2 < hp && hp <= 3) {
151 red = 0;
152 green = chroma;
153 blue = x;
154 }
155 else if (3 < hp && hp <= 4) {
156 red = 0;
157 green = x;
158 blue = chroma;
159 }
160 else if (4 < hp && hp <= 5) {
161 red = x;
162 green = 0;
163 blue = chroma;
164 }
165 else if (5 < hp && hp <= 6) {
166 red = chroma;
167 green = 0;
168 blue = x;
169 }
170
171 const auto m = v - chroma;
172
173 const auto r = static_cast<uint8>(std::round((red + m) * 255.0f));
174 const auto g = static_cast<uint8>(std::round((green + m) * 255.0f));
175 const auto b = static_cast<uint8>(std::round((blue + m) * 255.0f));
176
177 return {r, g, b};
178 }

191 [[nodiscard]] static auto from_hsl(float hue, float saturation, float lightness) -> color
192 {
193 hue = detail::clamp(hue, 0.0f, 360.0f);
194 saturation = detail::clamp(saturation, 0.0f, 100.0f);
195 lightness = detail::clamp(lightness, 0.0f, 100.0f);
196
197 const auto s = saturation / 100.0f;
198 const auto l = lightness / 100.0f;
199
200 const auto chroma = (1.0f - std::abs(2.0f * l - 1.0f)) * s;
201 const auto hp = hue / 60.0f;
202
203 const auto x = chroma * (1.0f - std::abs(std::fmod(hp, 2.0f) - 1.0f));
204
205 float red{};
206 float green{};
207 float blue{};
208
209 if (0 <= hp && hp < 1) {
210 red = chroma;
211 green = x;
212 blue = 0;
213 }
214 else if (1 <= hp && hp < 2) {
215 red = x;
216 green = chroma;
217 blue = 0;
218 }
219 else if (2 <= hp && hp < 3) {
220 red = 0;
221 green = chroma;
222 blue = x;
223 }
224 else if (3 <= hp && hp < 4) {
225 red = 0;
226 green = x;
227 blue = chroma;
228 }
229 else if (4 <= hp && hp < 5) {
230 red = x;
231 green = 0;
232 blue = chroma;
233 }
234 else if (5 <= hp && hp < 6) {
235 red = chroma;
236 green = 0;
237 blue = x;
238 }
239
240 const auto m = l - (chroma / 2.0f);
241
242 const auto r = static_cast<uint8>(std::round((red + m) * 255.0f));
243 const auto g = static_cast<uint8>(std::round((green + m) * 255.0f));
244 const auto b = static_cast<uint8>(std::round((blue + m) * 255.0f));
245
246 return {r, g, b};
247 }

261 [[nodiscard]] static auto from_rgb(const std::string_view rgb) -> std::optional<color>
262 {
263 if (rgb.length() != 7 || rgb.at(0) != '#') {
264 return std::nullopt;
265 }
266
267 const auto noHash = rgb.substr(1);
268
269 const auto rr = noHash.substr(0, 2);
270 const auto gg = noHash.substr(2, 2);
271 const auto bb = noHash.substr(4, 2);
272
273 const auto red = detail::stoi<uint8>(rr, 16);
274 const auto green = detail::stoi<uint8>(gg, 16);
275 const auto blue = detail::stoi<uint8>(bb, 16);
276
277 if (red && green && blue) {
278 return color{*red, *green, *blue};
279 }
280 else {
281 return std::nullopt;
282 }
283 }

297 [[nodiscard]] static auto from_rgba(const std::string_view rgba) -> std::optional<color>
298 {
299 if (rgba.length() != 9 || rgba.at(0) != '#') {
300 return std::nullopt;
301 }
302
303 const auto noHash = rgba.substr(1);
304
305 const auto rr = noHash.substr(0, 2);
306 const auto gg = noHash.substr(2, 2);
307 const auto bb = noHash.substr(4, 2);
308 const auto aa = noHash.substr(6, 2);
309
310 const auto red = detail::stoi<uint8>(rr, 16);
311 const auto green = detail::stoi<uint8>(gg, 16);
312 const auto blue = detail::stoi<uint8>(bb, 16);
313 const auto alpha = detail::stoi<uint8>(aa, 16);
314
315 if (red && green && blue && alpha) {
316 return color{*red, *green, *blue, *alpha};
317 }
318 else {
319 return std::nullopt;
320 }
321 }

335 [[nodiscard]] static auto from_argb(const std::string_view argb) -> std::optional<color>

337 if (argb.length() != 9 || argb.at(0) != '#') {

341 const auto noHash = argb.substr(1);

345 const auto gg = noHash.substr(4, 2);
346 const auto bb = noHash.substr(6, 2);
347
348 const auto alpha = detail::stoi<uint8>(aa, 16);
349 const auto red = detail::stoi<uint8>(rr, 16);
350 const auto green = detail::stoi<uint8>(gg, 16);
351 const auto blue = detail::stoi<uint8>(bb, 16);
352
353 if (alpha && red && green && blue) {
354 return color{*red, *green, *blue, *alpha};
355 }
356 else {
357 return std::nullopt;
358 }

361 [[nodiscard]] static auto from_norm(float red,
362 float green,
363 float blue,
364 float alpha = 1.0f) noexcept(on_msvc) -> color
365 {
366 red = detail::clamp(red, 0.0f, 1.0f);
367 green = detail::clamp(green, 0.0f, 1.0f);
368 blue = detail::clamp(blue, 0.0f, 1.0f);
369 alpha = detail::clamp(alpha, 0.0f, 1.0f);
370
371 const auto r = static_cast<uint8>(std::round(red * 255.0f));
372 const auto g = static_cast<uint8>(std::round(green * 255.0f));
373 const auto b = static_cast<uint8>(std::round(blue * 255.0f));
374 const auto a = static_cast<uint8>(std::round(alpha * 255.0f));
375
376 return color{r, g, b, a};
377 }

648[[nodiscard]] inline auto blend(const color& a, const color& b, const float bias = 0.5f)
649 -> color
650{
651 assert(bias >= 0);
652 assert(bias <= 1.0f);
653
654 const auto red = detail::lerp(a.norm_red(), b.norm_red(), bias);
655 const auto green = detail::lerp(a.norm_green(), b.norm_green(), bias);
656 const auto blue = detail::lerp(a.norm_blue(), b.norm_blue(), bias);
657 const auto alpha = detail::lerp(a.norm_alpha(), b.norm_alpha(), bias);
658
659 return color::from_norm(red, green, blue, alpha);
660}

672[[nodiscard]] constexpr auto operator==(const color& a, const color& b) noexcept -> bool
673{
674 return (a.red() == b.red()) && (a.green() == b.green()) && (a.blue() == b.blue()) &&
675 (a.alpha() == b.alpha());
676}

678[[nodiscard]] constexpr auto operator!=(const color& a, const color& b) noexcept -> bool
679{
680 return !(a == b);
