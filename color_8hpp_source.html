92 [[nodiscard]] static auto from_hsv(float hue, float saturation, float value) -> color
93 {
94 hue = detail::clamp(hue, 0.0f, 360.0f);
95 saturation = detail::clamp(saturation, 0.0f, 100.0f);
96 value = detail::clamp(value, 0.0f, 100.0f);
97
98 const auto v = (value / 100.0f);
99 const auto chroma = v * (saturation / 100.0f);
100 const auto hp = hue / 60.0f;
101
102 const auto x = chroma * (1.0f - std::abs(std::fmod(hp, 2.0f) - 1.0f));
103
104 float red{};
105 float green{};
106 float blue{};
107
108 if (0 <= hp && hp <= 1) {
109 red = chroma;
110 green = x;
111 blue = 0;
112 }
113 else if (1 < hp && hp <= 2) {
114 red = x;
115 green = chroma;
116 blue = 0;
117 }
118 else if (2 < hp && hp <= 3) {
119 red = 0;
120 green = chroma;
121 blue = x;
122 }
123 else if (3 < hp && hp <= 4) {
124 red = 0;
125 green = x;
126 blue = chroma;
127 }
128 else if (4 < hp && hp <= 5) {
129 red = x;
130 green = 0;
131 blue = chroma;
132 }
133 else if (5 < hp && hp <= 6) {
134 red = chroma;
135 green = 0;
136 blue = x;
137 }
138
139 const auto m = v - chroma;
140
141 const auto r = static_cast<uint8>(std::round((red + m) * 255.0f));
142 const auto g = static_cast<uint8>(std::round((green + m) * 255.0f));
143 const auto b = static_cast<uint8>(std::round((blue + m) * 255.0f));
144
145 return {r, g, b};
146 }

159 [[nodiscard]] static auto from_hsl(float hue, float saturation, float lightness) -> color
160 {
161 hue = detail::clamp(hue, 0.0f, 360.0f);
162 saturation = detail::clamp(saturation, 0.0f, 100.0f);
163 lightness = detail::clamp(lightness, 0.0f, 100.0f);
164
165 const auto s = saturation / 100.0f;
166 const auto l = lightness / 100.0f;
167
168 const auto chroma = (1.0f - std::abs(2.0f * l - 1.0f)) * s;
169 const auto hp = hue / 60.0f;
170
171 const auto x = chroma * (1.0f - std::abs(std::fmod(hp, 2.0f) - 1.0f));
172
173 float red{};
174 float green{};
175 float blue{};
176
177 if (0 <= hp && hp < 1) {
178 red = chroma;
179 green = x;
180 blue = 0;
181 }
182 else if (1 <= hp && hp < 2) {
183 red = x;
184 green = chroma;
185 blue = 0;
186 }
187 else if (2 <= hp && hp < 3) {
188 red = 0;
189 green = chroma;
190 blue = x;
191 }
192 else if (3 <= hp && hp < 4) {
193 red = 0;
194 green = x;
195 blue = chroma;
196 }
197 else if (4 <= hp && hp < 5) {
198 red = x;
199 green = 0;
200 blue = chroma;
201 }
202 else if (5 <= hp && hp < 6) {
203 red = chroma;
204 green = 0;
205 blue = x;
206 }
207
208 const auto m = l - (chroma / 2.0f);
209
210 const auto r = static_cast<uint8>(std::round((red + m) * 255.0f));
211 const auto g = static_cast<uint8>(std::round((green + m) * 255.0f));
212 const auto b = static_cast<uint8>(std::round((blue + m) * 255.0f));
213
214 return {r, g, b};
215 }

229 [[nodiscard]] static auto from_rgb(const std::string_view rgb) -> std::optional<color>
230 {
231 if (rgb.length() != 7 || rgb.at(0) != '#') {
232 return std::nullopt;
233 }
234
235 const auto noHash = rgb.substr(1);
236
237 const auto rr = noHash.substr(0, 2);
238 const auto gg = noHash.substr(2, 2);
239 const auto bb = noHash.substr(4, 2);
240
241 const auto red = detail::stoi<uint8>(rr, 16);
242 const auto green = detail::stoi<uint8>(gg, 16);
243 const auto blue = detail::stoi<uint8>(bb, 16);
244
245 if (red && green && blue) {
246 return color{*red, *green, *blue};
247 }
248 else {
249 return std::nullopt;
250 }
251 }

265 [[nodiscard]] static auto from_rgba(const std::string_view rgba) -> std::optional<color>
266 {
267 if (rgba.length() != 9 || rgba.at(0) != '#') {
268 return std::nullopt;
269 }
270
271 const auto noHash = rgba.substr(1);
272
273 const auto rr = noHash.substr(0, 2);
274 const auto gg = noHash.substr(2, 2);
275 const auto bb = noHash.substr(4, 2);
276 const auto aa = noHash.substr(6, 2);
277
278 const auto red = detail::stoi<uint8>(rr, 16);
279 const auto green = detail::stoi<uint8>(gg, 16);
280 const auto blue = detail::stoi<uint8>(bb, 16);
281 const auto alpha = detail::stoi<uint8>(aa, 16);
282
283 if (red && green && blue && alpha) {
284 return color{*red, *green, *blue, *alpha};
285 }
286 else {
287 return std::nullopt;
288 }
289 }

315
316 const auto alpha = detail::stoi<uint8>(aa, 16);
317 const auto red = detail::stoi<uint8>(rr, 16);
318 const auto green = detail::stoi<uint8>(gg, 16);
319 const auto blue = detail::stoi<uint8>(bb, 16);
320
321 if (alpha && red && green && blue) {
322 return color{*red, *green, *blue, *alpha};
323 }
324 else {
325 return std::nullopt;

329 [[nodiscard]] static auto from_norm(float red,
330 float green,
331 float blue,
332 float alpha = 1.0f) noexcept(on_msvc) -> color
333 {
334 red = detail::clamp(red, 0.0f, 1.0f);
335 green = detail::clamp(green, 0.0f, 1.0f);
336 blue = detail::clamp(blue, 0.0f, 1.0f);
337 alpha = detail::clamp(alpha, 0.0f, 1.0f);
338
339 const auto r = static_cast<uint8>(std::round(red * 255.0f));
340 const auto g = static_cast<uint8>(std::round(green * 255.0f));
341 const auto b = static_cast<uint8>(std::round(blue * 255.0f));
342 const auto a = static_cast<uint8>(std::round(alpha * 255.0f));
343
344 return color{r, g, b, a};
345 }

512[[nodiscard]] inline auto blend(const color& a, const color& b, const float bias = 0.5f)
513 -> color
514{
515 assert(bias >= 0);
516 assert(bias <= 1.0f);
517
518 const auto red = detail::lerp(a.norm_red(), b.norm_red(), bias);
519 const auto green = detail::lerp(a.norm_green(), b.norm_green(), bias);
520 const auto blue = detail::lerp(a.norm_blue(), b.norm_blue(), bias);
521 const auto alpha = detail::lerp(a.norm_alpha(), b.norm_alpha(), bias);
522
523 return color::from_norm(red, green, blue, alpha);
524}

536[[nodiscard]] constexpr auto operator==(const color& a, const color& b) noexcept -> bool
537{
538 return (a.red() == b.red()) && (a.green() == b.green()) && (a.blue() == b.blue()) &&
539 (a.alpha() == b.alpha());
