223 [[nodiscard]] auto convert_to(const pixel_format format) const -> surface
224 {
225 if (auto* converted = SDL_ConvertSurfaceFormat(mSurface, to_underlying(format), 0)) {
226 surface result{converted};
227 result.set_blend_mode(get_blend_mode());
228 return result;
229 }
230 else {
231 throw sdl_error{};
232 }
233 }

282 [[nodiscard]] auto color_mod() const noexcept -> color
283 {
284 uint8 red{};

359template <typename T>
360[[nodiscard]] auto to_string(const basic_surface<T>& surface) -> std::string
361{
362#if CENTURION_HAS_FEATURE_FORMAT
363 return std::format("surface(data: {}, width: {}, height: {})",
364 detail::address_of(surface.get()),
365 surface.width(),
366 surface.height());
367#else
368 return "surface(data: " + detail::address_of(surface.get()) +
369 ", width: " + std::to_string(surface.width()) +
370 ", height: " + std::to_string(surface.height()) + ")";
371#endif // CENTURION_HAS_FEATURE_FORMAT
372}
