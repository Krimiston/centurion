312 [[nodiscard]] auto convert_to(const pixel_format format) const -> surface
313 {
314 if (auto* converted = SDL_ConvertSurfaceFormat(mSurface, to_underlying(format), 0)) {
315 surface result{converted};
316 result.set_blend_mode(get_blend_mode());
317 return result;
318 }
319 else {
320 throw sdl_error{};
321 }

588template <typename T>
589[[nodiscard]] auto to_string(const basic_surface<T>& surface) -> std::string
590{
591#if CENTURION_HAS_FEATURE_FORMAT
592 return std::format("surface(data: {}, width: {}, height: {})",
593 detail::address_of(surface.get()),
594 surface.width(),
595 surface.height());
596#else
597 return "surface(data: " + detail::address_of(surface.get()) +
598 ", width: " + std::to_string(surface.width()) +
599 ", height: " + std::to_string(surface.height()) + ")";
600#endif // CENTURION_HAS_FEATURE_FORMAT
601}
