764template <typename T>
765[[nodiscard]] auto to_string(const basic_joystick<T>& joystick) -> std::string
766{
767 const char* serial{};
768 if constexpr (detail::sdl_version_at_least(2, 0, 14)) {
769 serial = joystick.serial();
770 }
771
772#if CENTURION_HAS_FEATURE_FORMAT
773 return std::format("joystick(data: {}, id: {}, name: '{}', serial: {})",
774 detail::address_of(joystick.get()),
775 joystick.GetID(),
776 str_or_na(joystick.GetName()),
777 str_or_na(serial));
778#else
779 return "joystick(data: " + detail::address_of(joystick.get()) +
780 ", id: " + std::to_string(joystick.id()) + ", name: '" + str_or_na(joystick.name()) +
781 ", serial: " + str_or_na(serial) + "')";
782#endif // CENTURION_HAS_FEATURE_FORMAT
