497[[nodiscard]] inline auto preferred_path(const char* org, const char* app) -> sdl_string
498{
499 /* Looking at the SDL source code, it actually seems fine to supply a null
500 string for the organization name. However, I haven't been able to find any
501 documentation providing this guarantee, so we simply disallow null
502 organization names. */
503 assert(org);
504 assert(app);
505 return sdl_string{SDL_GetPrefPath(org, app)};
506}
