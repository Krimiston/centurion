808[[nodiscard]] inline auto preferred_path(const char* org, const char* app) -> sdl_string
809{
810 /* Looking at the SDL source code, it actually seems fine to supply a null
811 string for the organization name. However, I haven't been able to find any
812 documentation providing this guarantee, so we simply disallow null
813 organization names. */
814 assert(org);
815 assert(app);
816 return sdl_string{SDL_GetPrefPath(org, app)};
