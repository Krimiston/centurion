57enum class sensor_type
58{
59 invalid = SDL_SENSOR_INVALID,
60 unknown = SDL_SENSOR_UNKNOWN,
61 accelerometer = SDL_SENSOR_ACCEL,
62 gyroscope = SDL_SENSOR_GYRO
63};
64
65[[nodiscard]] constexpr auto to_string(const sensor_type type) -> std::string_view
66{
67 switch (type) {
68 case sensor_type::invalid:
69 return "invalid";
70
71 case sensor_type::unknown:
72 return "unknown";
73
74 case sensor_type::accelerometer:
75 return "accelerometer";
76
77 case sensor_type::gyroscope:
78 return "gyroscope";
79
80 default:
81 throw exception{"Did not recognize sensor type!"};
82 }
83}

85inline auto operator<<(std::ostream& stream, const sensor_type type) -> std::ostream&
86{

244{

251 return "sensor(data: " + detail::address_of(sensor.get()) +
