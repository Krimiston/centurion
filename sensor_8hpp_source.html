399[[nodiscard]] auto to_string(const basic_sensor<T>& sensor) -> std::string
400{
401#if CENTURION_HAS_FEATURE_FORMAT
402 return std::format("sensor(data: {}, id: {}, name: '{}')",
403 detail::address_of(sensor.get()),
404 sensor.id().value(),
405 str_or_na(sensor.name()));
406#else
407 return "sensor(data: " + detail::address_of(sensor.get()) +
408 ", id: " + std::to_string(sensor.id().value()) + ", name: '" +
409 str_or_na(sensor.name()) + "')";
